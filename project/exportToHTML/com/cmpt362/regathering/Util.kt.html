18   object Util {
19       fun checkPermissions(activity: Activity?) {
20           if (Build.VERSION.SDK_INT < 23) return
21           if (ContextCompat.checkSelfPermission(
22                   activity!!,
23                   Manifest.permission.WRITE_EXTERNAL_STORAGE
24               )
25               != PackageManager.PERMISSION_GRANTED
26               || ContextCompat.checkSelfPermission(activity, Manifest.permission.CAMERA)
27               != PackageManager.PERMISSION_GRANTED) {
28               ActivityCompat.requestPermissions(
29                   activity,
30                   arrayOf(Manifest.permission.WRITE_EXTERNAL_STORAGE, Manifest.permission.CAMERA),
31                   0
32               )
33           }
34       }
35   
36       fun getBitmap(context: Context, imgUri: Uri): Bitmap {
37           var bitmap = BitmapFactory.decodeStream(context.contentResolver.openInputStream(imgUri))
38           val matrix = Matrix()
39           matrix.setRotate(90f)
40           var ret = Bitmap.createBitmap(bitmap, 0, 0, bitmap.width, bitmap.height, matrix, true)
41           return ret
42       }
