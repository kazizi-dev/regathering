55       override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
56           super.onViewCreated(view, savedInstanceState)
57           getEventsFromDatabase("hostedEvents")
58           binding.switchViewEvents.text = "Hosted Events"
59           binding.switchViewEvents.setOnClickListener(){
60               hostedEvents = !hostedEvents
61               if(hostedEvents){
62                   binding.switchViewEvents.text = "Hosted Events"
63                   binding.upcomingEvents.text = "Hosted Events"
64                   getEventsFromDatabase("hostedEvents")
65               }
66               else{
67                   binding.switchViewEvents.text = "Joined Events"
68                   binding.upcomingEvents.text = "Joined Events"
69                   getEventsFromDatabase("joinedEvents")
70               }

76       private fun getEventsFromDatabase(eventsType: String){
77           firestore = Firebase.firestore
78           val user = FirebaseAuth.getInstance().currentUser!!.uid
79           firestore.collection("users").document(user).get().addOnSuccessListener {
80               println("debug: user $it")
81               val hostedEvents = ArrayList<String>()
82               hostedEvents.addAll(it.get(eventsType) as Collection<String>)
83               println("debug: hostedEvents $hostedEvents")
84   
85               if(hostedEvents.isNotEmpty()){
86                   eventsQuery = firestore.collection("events")
87                       .whereIn(FieldPath.documentId(), hostedEvents)
88                       .limit(LIMIT.toLong())
89                   // RecyclerView
90                   eventAdapter = object : EventAdapter(eventsQuery, this@FragmentHome) {
91                       override fun onDataChanged() {
92                           // Show/hide content if the query returns empty.
93                           if (itemCount == 0) {
94                               binding.recyclerEvents.visibility = View.GONE
95                           } else {
96                               binding.recyclerEvents.visibility = View.VISIBLE
97                           }
98                       }
99   
100                      override fun onError(e: FirebaseFirestoreException) {
101                          // Show a snackbar on errors
102                          Snackbar.make(binding.root,
103                              "Error: check logs for info.", Snackbar.LENGTH_LONG).show()
104                      }
105                  }
106  
107                  val layoutManager = LinearLayoutManager(context)
108                  binding.recyclerEvents.layoutManager = layoutManager
109                  binding.recyclerEvents.adapter = eventAdapter
110                  binding.recyclerEvents.addItemDecoration(
111                      DividerItemDecoration(
112                          binding.recyclerEvents.context,
113                          LinearLayoutManager.HORIZONTAL
114                      ))
115                  eventAdapter.startListening()
116              }
117              else{
118                  val layoutManager = LinearLayoutManager(context)
119                  binding.recyclerEvents.layoutManager = layoutManager
120                  binding.recyclerEvents.visibility = View.GONE
121              }
122          }
123      }

143      override fun onEventSelected(event: DocumentSnapshot) {
144          // TODO: Go to the details page for the selected restaurant
145          println("DEBUG: Selected event -> $event")
146          if(hostedEvents){
147              val builder: AlertDialog.Builder = AlertDialog.Builder(requireContext())
148              builder.setTitle("Cancel Event?")
149  
150              builder.setPositiveButton("Yes") { dialog, which ->
151                  val tempId = event.id
152                  firestore.collection("events").document(event.id)
153                      .delete()
154                      .addOnSuccessListener{Log.d(TAG,"DocumentSnapshot successfully deleted!")}
155                      .addOnSuccessListener{e->Log.w(TAG, "Error deleting document: $e")}
156                  firestore.collection("users").document(FirebaseAuth.getInstance().currentUser!!.uid)
157                      .get()
158                      .addOnSuccessListener {
159                          it.reference.update("hostedEvents", FieldValue.arrayRemove(tempId))
160                      }
161                  firestore.collection("users").get().addOnSuccessListener {
162                      for(document in it.documents){
163                          //val arrayOfJoinedEvents = document.get("joinedEvents") as ArrayList<*>
164                          document.reference.update("joinedEvents", FieldValue.arrayRemove(tempId))
165                      }
166                  }
167              }
168              builder.setNegativeButton("No"){dialog, which -> }
169              builder.show()
170          }
171          else{
172              val builder: AlertDialog.Builder = AlertDialog.Builder(requireContext())
173              builder.setTitle("Cancel Attendance?")
174  
175              builder.setPositiveButton("Yes") { dialog, which ->
176                  firestore.collection("users").document(FirebaseAuth.getInstance().currentUser!!.uid)
177                      .get()
178                      .addOnSuccessListener {
179                          it.reference.update("joinedEvents", FieldValue.arrayRemove(event.id))
180                          getEventsFromDatabase("joinedEvents")
181                      }
182              }
183              builder.setNegativeButton("No"){dialog, which -> }
184              builder.show()
185          }
186      }
